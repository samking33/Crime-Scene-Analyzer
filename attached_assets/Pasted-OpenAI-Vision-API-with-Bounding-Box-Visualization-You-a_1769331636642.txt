2. Image Annotation Service (Backend):
Create a service to draw bounding boxes using Canvas:
javascript// services/imageAnnotationService.js
const { createCanvas, loadImage } = require('canvas');

// Category color mapping
const CATEGORY_COLORS = {
  weapons: '#D32F2F',      // Red
  vehicles: '#F57C00',     // Orange
  persons: '#1976D2',      // Blue
  biometrics: '#7B1FA2',   // Purple
  drugs: '#C62828',        // Dark Red
  documents: '#388E3C',    // Green
  electronics: '#00796B',  // Teal
  evidenceMarkers: '#FBC02D', // Yellow
  tools: '#5D4037',        // Brown
  other: '#616161'         // Gray
};

async function drawBoundingBoxes(base64Image, detectedObjects) {
  try {
    // Convert base64 to buffer
    const imageBuffer = Buffer.from(base64Image, 'base64');
    
    // Load image
    const img = await loadImage(imageBuffer);
    
    // Create canvas with same dimensions
    const canvas = createCanvas(img.width, img.height);
    const ctx = canvas.getContext('2d');
    
    // Draw original image
    ctx.drawImage(img, 0, 0);
    
    // Draw bounding boxes for each detected object
    detectedObjects.forEach((obj, index) => {
      const { boundingBox, category, name, confidence } = obj;
      
      // Convert normalized coordinates to pixel coordinates
      const x = boundingBox.x * img.width;
      const y = boundingBox.y * img.height;
      const width = boundingBox.width * img.width;
      const height = boundingBox.height * img.height;
      
      // Get color for category
      const color = CATEGORY_COLORS[category] || CATEGORY_COLORS.other;
      
      // Draw semi-transparent rectangle
      ctx.strokeStyle = color;
      ctx.lineWidth = 4;
      ctx.strokeRect(x, y, width, height);
      
      // Draw filled background for label
      ctx.fillStyle = color;
      ctx.globalAlpha = 0.8;
      const labelText = `${name} - ${confidence}`;
      const textMetrics = ctx.measureText(labelText);
      const labelHeight = 24;
      const labelWidth = textMetrics.width + 16;
      
      ctx.fillRect(x, y - labelHeight, labelWidth, labelHeight);
      
      // Draw label text
      ctx.globalAlpha = 1.0;
      ctx.fillStyle = '#FFFFFF';
      ctx.font = 'bold 14px Arial';
      ctx.fillText(labelText, x + 8, y - 8);
      
      // Draw object number badge
      ctx.fillStyle = color;
      ctx.beginPath();
      ctx.arc(x - 12, y - 12, 16, 0, 2 * Math.PI);
      ctx.fill();
      ctx.fillStyle = '#FFFFFF';
      ctx.font = 'bold 12px Arial';
      ctx.textAlign = 'center';
      ctx.fillText((index + 1).toString(), x - 12, y - 7);
      ctx.textAlign = 'left';
    });
    
    // Convert canvas to base64
    const annotatedBase64 = canvas.toBuffer('image/jpeg', { quality: 0.95 }).toString('base64');
    
    return annotatedBase64;
    
  } catch (error) {
    console.error('Error drawing bounding boxes:', error);
    throw error;
  }
}

module.exports = { drawBoundingBoxes };

Modify the TanStack Query mutation to save both images:
javascript// hooks/useEvidenceAnalysis.js
import { useMutation } from '@tanstack/react-query';
import * as FileSystem from 'expo-file-system';

export const useAnalyzeEvidence = () => {
  return useMutation({
    mutationFn: async ({ imageUri, caseId, evidenceId }) => {
      // Read image as base64
      const base64 = await FileSystem.readAsStringAsync(imageUri, {
        encoding: FileSystem.EncodingType.Base64,
      });
      
      // Send to backend
      const response = await fetch('http://localhost:5000/api/analyze-evidence', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ 
          image: base64, 
          caseId, 
          evidenceId 
        }),
      });
      
      const data = await response.json();
      
      // Save annotated image to local filesystem
      const annotatedImagePath = `${FileSystem.documentDirectory}${caseId}/annotated_${evidenceId}.jpg`;
      await FileSystem.writeAsStringAsync(annotatedImagePath, data.annotatedImage, {
        encoding: FileSystem.EncodingType.Base64,
      });
      
      return {
        ...data,
        annotatedImageUri: annotatedImagePath,
        originalImageUri: imageUri,
      };
    },
    onSuccess: async (data, { caseId, evidenceId }) => {
      // Update evidence in AsyncStorage with both image URIs
      const existingData = await AsyncStorage.getItem(`evidence_${caseId}`);
      const evidenceList = existingData ? JSON.parse(existingData) : [];
      
      const updatedEvidence = evidenceList.map(evidence => {
        if (evidence.id === evidenceId) {
          return {
            ...evidence,
            originalImageUri: data.originalImageUri,
            annotatedImageUri: data.annotatedImageUri,
            detectedObjects: data.objects,
            aiSummary: data.summary,
            detectionStatus: 'completed',
          };
        }
        return evidence;
      });
      
      await AsyncStorage.setItem(`evidence_${caseId}`, JSON.stringify(updatedEvidence));
    },
  });
};

javascript// screens/EvidenceDetailScreen.js
import { View, Image, Dimensions, Text, ScrollView, TouchableOpacity } from 'react-native';
import { useState } from 'react';
import { GestureDetector, Gesture } from 'react-native-gesture-handler';
import Animated, { useSharedValue, useAnimatedStyle, withTiming } from 'react-native-reanimated';

const { width } = Dimensions.get('window');

export default function EvidenceDetailScreen({ route }) {
  const { evidence } = route.params;
  const [viewMode, setViewMode] = useState('annotated'); // 'annotated' | 'original' | 'split'
  
  const translateX = useSharedValue(0);
  
  const swipe = Gesture.Pan()
    .onUpdate((e) => {
      translateX.value = e.translationX;
    })
    .onEnd(() => {
      if (translateX.value < -50 && viewMode === 'original') {
        setViewMode('annotated');
      } else if (translateX.value > 50 && viewMode === 'annotated') {
        setViewMode('original');
      }
      translateX.value = withTiming(0);
    });
  
  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: translateX.value }],
  }));
  
  return (
    <ScrollView style={styles.container}>
      {/* View Mode Selector */}
      <View style={styles.modeSelector}>
        <TouchableOpacity
          style={[styles.modeButton, viewMode === 'original' && styles.modeButtonActive]}
          onPress={() => setViewMode('original')}
        >
          <Text style={styles.modeButtonText}>Original</Text>
        </TouchableOpacity>
        
        <TouchableOpacity
          style={[styles.modeButton, viewMode === 'annotated' && styles.modeButtonActive]}
          onPress={() => setViewMode('annotated')}
        >
          <Text style={styles.modeButtonText}>Annotated</Text>
        </TouchableOpacity>
        
        <TouchableOpacity
          style={[styles.modeButton, viewMode === 'split' && styles.modeButtonActive]}
          onPress={() => setViewMode('split')}
        >
          <Text style={styles.modeButtonText}>Split View</Text>
        </TouchableOpacity>
      </View>
      
      {/* Image Display */}
      {viewMode === 'split' ? (
        <View style={styles.splitContainer}>
          <View style={styles.splitHalf}>
            <Text style={styles.splitLabel}>Original</Text>
            <Image 
              source={{ uri: evidence.originalImageUri }}
              style={styles.splitImage}
              resizeMode="contain"
            />
          </View>
          <View style={styles.splitHalf}>
            <Text style={styles.splitLabel}>Annotated</Text>
            <Image 
              source={{ uri: evidence.annotatedImageUri }}
              style={styles.splitImage}
              resizeMode="contain"
            />
          </View>
        </View>
      ) : (
        <GestureDetector gesture={swipe}>
          <Animated.View style={animatedStyle}>
            <Image
              source={{ 
                uri: viewMode === 'annotated' 
                  ? evidence.annotatedImageUri 
                  : evidence.originalImageUri 
              }}
              style={styles.fullImage}
              resizeMode="contain"
            />
            <Text style={styles.swipeHint}>
              ← Swipe to toggle between views →
            </Text>
          </Animated.View>
        </GestureDetector>
      )}
      
      {/* AI Analysis Summary */}
      {evidence.aiSummary && (
        <View style={styles.summaryCard}>
          <Text style={styles.summaryTitle}>AI Analysis Summary</Text>
          <Text style={styles.summaryText}>{evidence.aiSummary}</Text>
        </View>
      )}
      
      {/* Detected Objects List */}
      {evidence.detectedObjects?.length > 0 && (
        <View style={styles.objectsCard}>
          <Text style={styles.objectsTitle}>
            Detected Objects ({evidence.detectedObjects.length})
          </Text>
          {evidence.detectedObjects.map((obj, index) => (
            <View key={obj.id} style={styles.objectItem}>
              <View style={[styles.objectBadge, { backgroundColor: getCategoryColor(obj.category) }]}>
                <Text style={styles.objectNumber}>{index + 1}</Text>
              </View>
              <View style={styles.objectInfo}>
                <Text style={styles.objectName}>{obj.name}</Text>
                <Text style={styles.objectDetails}>
                  {obj.category} • {obj.confidence} confidence
                </Text>
              </View>
            </View>
          ))}
        </View>
      )}
    </ScrollView>
  );
}
