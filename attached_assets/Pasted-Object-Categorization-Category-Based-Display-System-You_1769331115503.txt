const categoryKeywords = {
  1: ['gun', 'pistol', 'rifle', 'knife', 'blade', 'weapon', 'firearm', 'ammunition', 'bullet', 'shell casing', 'revolver', 'shotgun'],
  2: ['car', 'truck', 'van', 'motorcycle', 'bicycle', 'vehicle', 'SUV', 'sedan', 'license plate', 'tire', 'windshield'],
  3: ['person', 'individual', 'suspect', 'victim', 'witness', 'face', 'body', 'human', 'male', 'female'],
  4: ['blood', 'fingerprint', 'DNA', 'bodily fluid', 'footprint', 'shoe print', 'hair', 'saliva', 'tissue'],
  5: ['drugs', 'narcotics', 'pills', 'powder', 'syringe', 'paraphernalia', 'marijuana', 'cocaine', 'heroin', 'meth'],
  6: ['document', 'paper', 'ID card', 'passport', 'receipt', 'letter', 'note', 'contract', 'certificate', 'license'],
  7: ['phone', 'computer', 'laptop', 'tablet', 'camera', 'USB', 'hard drive', 'SIM card', 'charger', 'battery'],
  8: ['evidence marker', 'cone', 'flag', 'numbered marker', 'tape', 'barrier', 'crime scene tape'],
  9: ['hammer', 'screwdriver', 'crowbar', 'lockpick', 'tool', 'pliers', 'wrench', 'drill'],
  10: [] // Other - default
};

function categorizeObject(objectName) {
  const nameLower = objectName.toLowerCase();
  
  // Find matching category
  for (const [categoryId, keywords] of Object.entries(categoryKeywords)) {
    if (keywords.some(keyword => nameLower.includes(keyword))) {
      return parseInt(categoryId);
    }
  }
  
  return 10; // Default to "Other"
}

module.exports = { categorizeObject };
5. Frontend Data Fetching with TanStack Query:
javascript// hooks/useEvidenceAnalysis.js
import { useMutation } from '@tanstack/react-query';

export const useAnalyzeEvidence = () => {
  return useMutation({
    mutationFn: async ({ imageUri, caseId }) => {
      // Convert image to base64
      const base64 = await FileSystem.readAsStringAsync(imageUri, {
        encoding: FileSystem.EncodingType.Base64,
      });
      
      // Send to backend
      const response = await fetch('http://localhost:5000/api/analyze-evidence', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ image: base64, caseId }),
      });
      
      return response.json();
    },
    onSuccess: async (data, { caseId, evidenceId }) => {
      // Store categorized objects in AsyncStorage
      const existingData = await AsyncStorage.getItem(`detectedObjects_${caseId}`);
      const objects = existingData ? JSON.parse(existingData) : { caseId, objects: [] };
      
      // Add new objects with unique IDs
      const newObjects = data.objects.map(obj => ({
        ...obj,
        id: crypto.randomUUID(),
        evidenceId,
        detectedAt: Date.now(),
      }));
      
      objects.objects.push(...newObjects);
      await AsyncStorage.setItem(`detectedObjects_${caseId}`, JSON.stringify(objects));
    },
  });
};

import { ScrollView, View, Text, TouchableOpacity } from 'react-native';

export default function EvidencePhotoView({ evidenceId, caseId }) {
  const { data: detectedObjects } = useQuery({
    queryKey: ['evidenceObjects', evidenceId],
    queryFn: async () => {
      const data = await AsyncStorage.getItem(`detectedObjects_${caseId}`);
      if (!data) return [];
      
      const { objects } = JSON.parse(data);
      return objects.filter(obj => obj.evidenceId === evidenceId);
    },
  });
  
  // Group objects by category
  const groupedByCategory = detectedObjects?.reduce((acc, obj) => {
    if (!acc[obj.categoryName]) acc[obj.categoryName] = [];
    acc[obj.categoryName].push(obj);
    return acc;
  }, {});
  
  return (
    <View>
      {/* Category Chips */}
      <ScrollView horizontal showsHorizontalScrollIndicator={false}>
        {Object.keys(groupedByCategory || {}).map(categoryName => (
          <TouchableOpacity

9. Category Statistics with Charts:
javascript// components/CategoryAnalytics.js
import { PieChart } from 'react-native-chart-kit';
import { Dimensions } from 'react-native';

export default function CategoryAnalytics({ caseId }) {
  const { data: stats } = useQuery({
    queryKey: ['categoryAnalytics', caseId],
    queryFn: async () => {
      const data = await AsyncStorage.getItem(`detectedObjects_${caseId}`);
      if (!data) return null;
      
      const { objects } = JSON.parse(data);
      
      // Calculate distribution
      const distribution = {};
      objects.forEach(obj => {
        distribution[obj.categoryName] = (distribution[obj.categoryName] || 0) + 1;
      });
      
      return {
        mostCommon: Object.entries(distribution).sort((a, b) => b[1] - a[1])[0],
        distribution,
        highPriorityAlerts: objects.filter(obj => obj.categoryId <= 2).length,
      };
    },
  });
  
  const chartData = Object.entries(stats?.distribution || {}).map(([name, count], index) => ({
    name,
    count,
    color: categories.find(c => c.categoryName === name.toLowerCase())?.colorCode || '#999',
    legendFontColor: '#FFF',
    legendFontSize: 12,
  }));
  
  return (
    <View>
      <Text style={styles.statTitle}>Most Common Category</Text>
      <Text style={styles.statValue}>
        {stats?.mostCommon?.[0]} ({stats?.mostCommon?.[1]} detections)
      </Text>
      
      {stats?.highPriorityAlerts > 0 && (
        <View style={styles.alert}>
          <Text style={styles.alertText}>
            ⚠️ {stats.highPriorityAlerts} high-priority items detected
          </Text>
        </View>
      )}
      
      <PieChart
        data={chartData}
        width={Dimensions.get('window').width - 32}
        height={220}
        chartConfig={{
          color: (opacity = 1) => `rgba(255, 255, 255, ${opacity})`,
        }}
        accessor="count"
        backgroundColor="transparent"
        paddingLeft="15"
      />
    </View>
  );
}

javascript// utils/generatePDFReport.js
import * as Print from 'expo-print';
import * as Sharing from 'expo-sharing';
import { format } from 'date-fns';

export async function generateCategoryReport(caseId) {
  // Fetch all data
  const data = await AsyncStorage.getItem(`detectedObjects_${caseId}`);
  const { objects } = JSON.parse(data);
  
  // Group by category
  const byCategory = objects.reduce((acc, obj) => {
    if (!acc[obj.categoryName]) acc[obj.categoryName] = [];
    acc[obj.categoryName].push(obj);
    return acc;
  }, {});
  
  // Generate HTML
  const html = `
    <html>
      <head>
        <style>
          body { font-family: Arial, sans-serif; padding: 20px; }
          .category-section { margin-bottom: 30px; page-break-inside: avoid; }
          .category-header { 
            color: #333; 
            border-left: 4px solid;
            padding-left: 12px;
            margin-bottom: 12px;
          }
          table { width: 100%; border-collapse: collapse; }
          th, td { border: 1px solid #ddd; padding: 8px; text-align: left; }
          th { background-color: #f2f2f2; }
          .high-priority { background-color: #ffebee; }
        </style>
      </head>
      <body>
        <h1>Evidence by Category</h1>
        <p>Case ID: ${caseId}</p>
        <p>Generated: ${format(new Date(), 'PPpp')}</p>
        
        ${Object.entries(byCategory).map(([categoryName, objects]) => {
          const category = categories.find(c => c.categoryName === categoryName);
          return `
            <div class="category-section ${category.priority <= 2 ? 'high-priority' : ''}">
              <h2 class="category-header" style="border-left-color: ${category.colorCode}">
                ${category.displayName} (${objects.length})
              </h2>
              <table>
                <tr>
                  <th>Object</th>
                  <th>Confidence</th>
                  <th>Location</th>
                  <th>Detected At</th>
                </tr>
                ${objects.map(obj => `
                  <tr>
                    <td>${obj.objectName}</td>
                    <td>${obj.confidence}</td>
                    <td>${obj.location}</td>
                    <td>${format(obj.detectedAt, 'PPpp')}</td>
                  </tr>
                `).join('')}
              </table>
            </div>
          `;
        }).join('')}
      </body>
    </html>
  `;
  
  // Generate PDF
  const { uri } = await Print.printToFileAsync({ html });
  
  // Share
  await Sharing.shareAsync(uri);
}
